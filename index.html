💸 송금하기  Transaction

트랜잭션 실행 흐름 (송금 버튼 클릭 시) -- server.js / POST /api/transfer
BEGIN;
  -- ① 잔액 확인 + 행 잠금 (동시 송금 충돌 방지)
  SELECT balance FROM accounts WHERE id = $fromId FOR UPDATE;
  -- ② 송금자 차감
  UPDATE accounts SET balance = balance - $amount WHERE id = $fromId;
  -- ③ 수신자 증가
  UPDATE accounts SET balance = balance + $amount WHERE id = $toId;
  -- ④ 송금 이력 기록 (transfers 테이블 INSERT)
  INSERT INTO transfers (from_account_id, to_account_id, amount)
      VALUES ($fromId, $toId, $amount);
COMMIT;  -- 전부 성공 시 / 실패 시 → ROLLBACK;
